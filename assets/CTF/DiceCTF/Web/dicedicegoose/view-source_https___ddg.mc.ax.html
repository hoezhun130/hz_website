function encode(history) {
const data = new Uint8Array(history.length * 4);

let idx = 0;
for (const part of history) {
data[idx++] = part[0][0];
data[idx++] = part[0][1];
data[idx++] = part[1][0];
data[idx++] = part[1][1];
}

let prev = String.fromCharCode.apply(null, data);
let ret = btoa(prev);
return ret;
}
